130 uros_err_t uros_tcpros_sendheader(UrosTcpRosStatus *tcpstp,
131  uros_bool_t thisisclient) {
132 
133  size_t hdrlen = 0;
134  uint32_t hdrlen32;
135  const UrosString *calleridstrp, *namestrp, *typestrp, *md5strp;
136 
137  urosAssert(tcpstp != NULL);
138  urosAssert(tcpstp->topicp != NULL);
139  urosAssert(tcpstp->topicp->typep != NULL);
140 #define _CHKOK { if (tcpstp->err != UROS_OK) { return tcpstp->err; } }
141 
142  calleridstrp = &urosNode.config.nodeName;
143  typestrp = &tcpstp->topicp->typep->name;
144  md5strp = &tcpstp->topicp->typep->md5str;
145 
146  hdrlen += 4 + calleridfield.length + 1 + calleridstrp->length;
147  hdrlen += 4 + md5field.length + 1 + md5strp->length;
148  hdrlen += 4 + typefield.length + 1 + typestrp->length;
149  if (thisisclient) {
150  namestrp = &tcpstp->topicp->name;
151  if (tcpstp->flags.service) {
152  hdrlen += 4 + servicefield.length + 1 + namestrp->length;
153  hdrlen += 4 + persistentfield.length + 2;
154  } else {
155  hdrlen += 4 + topicfield.length + 1 + namestrp->length;
156  hdrlen += 4 + tcpnodelayfield.length + 2;
157  }
158  } else {
159  namestrp = NULL;
160  if (!tcpstp->flags.service) {
161  hdrlen += 4 + latchingfield.length + 2;
162  }
163  }
164 
165  if (tcpstp->flags.service) {
166  /* Write the service ok byte.*/
167  uint8_t ok = 1;
168  urosTcpRosSend(tcpstp, &ok, sizeof(ok)); _CHKOK
169  }
170 
171  /* uint32 header_length */
172  hdrlen32 = (uint32_t)hdrlen;
173  urosTcpRosSend(tcpstp, &hdrlen, sizeof(hdrlen32)); _CHKOK
174 
175  /* uint32 field_length, callerid={str} */
176  uros_tcpros_sendfieldstring(tcpstp, &calleridfield, calleridstrp); _CHKOK
177 
178  if (thisisclient) {
179  if (tcpstp->flags.service) {
180  /* uint32 field_length, service={str} */
181  uros_tcpros_sendfieldstring(tcpstp, &servicefield, namestrp); _CHKOK
182  } else {
183  /* uint32 field_length, topic={str} */
184  uros_tcpros_sendfieldstring(tcpstp, &topicfield, namestrp); _CHKOK
185  }
186  }
187 
188  /* uint32 field_length, md5sum={str} */
189  uros_tcpros_sendfieldstring(tcpstp, &md5field, md5strp); _CHKOK
190 
191  /* uint32 field_length, type={str} */
192  uros_tcpros_sendfieldstring(tcpstp, &typefield, typestrp); _CHKOK
193 
194  if (thisisclient) {
195  if (tcpstp->flags.service) {
196  /* uint32 field_length, persistent=(0|1) */
197  uros_tcpros_sendfieldbool(tcpstp, &persistentfield,
198  tcpstp->flags.persistent); _CHKOK
199  } else {
200  /* uint32 field_length, tcp_nodelay=(0|1) */
201  uros_tcpros_sendfieldbool(tcpstp, &tcpnodelayfield,
202  tcpstp->flags.noDelay); _CHKOK
203  }
204  } else {
205  if (!tcpstp->flags.service) {
206  /* uint32 field_length, latching=(0|1) */
207  uros_tcpros_sendfieldbool(tcpstp, &latchingfield,
208  tcpstp->flags.latching); _CHKOK
209  }
210  }
211 
212  return tcpstp->err = UROS_OK;
213 #undef _CHKOK
214 }

216 uros_err_t uros_tcpros_recvheader(UrosTcpRosStatus *tcpstp,
217  uros_bool_t thisisclient) {
218 
219  uint32_t hdrlen, remlen;
220  uint32_t vallen, fieldlen;
221  char *valp = NULL;
222  UrosTopic *topicp;
223  UrosMsgType *typep;
224  char buf[4]; /* Length matches the shortest field ("type").*/
225 
226  urosAssert(tcpstp != NULL);
227 #define _CHKOK { if (tcpstp->err != UROS_OK) { goto _error; } }
228 #define _GOT(tokp, toklen) \
229  ((toklen <= 4) \
230  ? (0 == memcmp(buf, tokp, toklen)) \
231  : (0 == memcmp(buf, tokp, 4) && \
232  UROS_OK == urosTcpRosExpect(tcpstp, tokp + 4, toklen - 4)))
233 
234  if (!thisisclient) {
235  urosAssert(tcpstp->topicp == NULL);
236  typep = urosNew(UrosMsgType);
237  urosMsgTypeObjectInit(typep);
238  topicp = urosNew(UrosTopic);
239  urosTopicObjectInit(topicp);
240  topicp->typep = typep;
241  } else {
242  urosAssert(tcpstp->topicp != NULL);
243  urosAssert(tcpstp->topicp->typep != NULL);
244  topicp = (UrosTopic*)tcpstp->topicp;
245  typep = (UrosMsgType*)topicp->typep;
246  }
247 
248  /* Read the header length.*/
249  urosTcpRosRecv(tcpstp, &hdrlen, sizeof(hdrlen)); _CHKOK
250 
251  /* Get each header field.*/
252  for (remlen = hdrlen; remlen > 0; remlen -= fieldlen) {
253  UrosString *strp = NULL;
254 
255  /* Get the field length and check for size consistency.*/
256  if (remlen < 4) { tcpstp->err = UROS_ERR_BADCONN; goto _error; }
257  remlen -= 4;
258  urosTcpRosRecv(tcpstp, (char*)&fieldlen, sizeof(fieldlen)); _CHKOK
259  if (remlen < fieldlen) { tcpstp->err = UROS_ERR_BADCONN; goto _error; }
260  vallen = fieldlen;
261 
262  /* Decode the field name.*/
263  urosTcpRosRecv(tcpstp, buf, 4); /* Minimum-length field.*/
264  if (_GOT("callerid=", 9)) {
265  /* callerid={str} */
266  strp = &tcpstp->callerId;
267  vallen -= 9;
268  } else if (_GOT("error=", 6)) {
269  /* error={str} */
270  strp = &tcpstp->errstr;
271  vallen -= 6;
272  } else if (_GOT("latching=", 9)) {
273  if (!thisisclient) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
274  if (tcpstp->flags.service) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
275  /* latching=(0|1) */
276  if (vallen == 10) {
277  urosTcpRosRecv(tcpstp, buf, 1);
278  if (buf[0] == '1') { tcpstp->flags.latching = UROS_TRUE; continue; }
279  else if (buf[0] == '0') { tcpstp->flags.latching = UROS_FALSE; continue; }
280  else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
281  } else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
282  } else if (_GOT("md5sum=", 7)) {
283  /* md5sum={str} */
284  strp = &typep->md5str;
285  vallen -= 7;
286  if (thisisclient) {
287  /* Check if it matches the referenced one.*/
288  if (vallen != strp->length) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
289  urosTcpRosExpect(tcpstp, strp->datap, strp->length);
290  if (tcpstp->err == UROS_OK) { continue; }
291  else { goto _error; }
292  }
293  } else if (_GOT("message_definition=", 19)) {
294  /* message_definition={str} */
295  vallen -= 19;
296 #if UROS_TCPROS_USE_MESSAGE_DEFINITION
297  strp = &typep->desc;
298 #else
299  urosTcpRosSkip(tcpstp, vallen);
300  continue;
301 #endif
302  } else if (_GOT("persistent=", 11)) {
303  if (thisisclient) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
304  if (!tcpstp->flags.service) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
305  /* persistent=(0|1) */
306  if (vallen == 12) {
307  urosTcpRosRecv(tcpstp, buf, 1);
308  if (buf[0] == '1') { tcpstp->flags.persistent = UROS_TRUE; continue; }
309  else if (buf[0] == '0') { tcpstp->flags.persistent = UROS_FALSE; continue; }
310  else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
311  } else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
312  } else if (_GOT("probe=", 6)) {
313  /* probe=(0|1) */
314  if (vallen == 7) {
315  urosTcpRosRecv(tcpstp, buf, 1);
316  if (buf[0] == '1') { tcpstp->flags.probe = UROS_TRUE; continue; }
317  else if (buf[0] == '0') { tcpstp->flags.probe = UROS_FALSE; continue; }
318  else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
319  } else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
320  } else if (_GOT("service=", 8)) {
321  if (thisisclient) {
322  /* Check if it matches the referenced one.*/
323  urosTcpRosExpect(tcpstp, topicp->name.datap, topicp->name.length);
324  if (tcpstp->err == UROS_OK) { continue; }
325  else { goto _error; }
326  } else {
327  /* service={str} */
328  strp = &topicp->name;
329  tcpstp->flags.service = UROS_TRUE;
330  vallen -= 8;
331  }
332  } else if (_GOT("topic=", 6)) {
333  if (thisisclient) {
334  /* Check if it matches the referenced one.*/
335  urosTcpRosExpect(tcpstp, topicp->name.datap, topicp->name.length);
336  if (tcpstp->err == UROS_OK) { continue; }
337  else { goto _error; }
338  } else {
339  /* topic={str} */
340  strp = &topicp->name;
341  tcpstp->flags.service = UROS_FALSE;
342  vallen -= 6;
343  }
344  } else if (_GOT("type=", 5)) { /* <<< Minimum-length field.*/
345  /* type={str} */
346  strp = &typep->name;
347  vallen -= 5;
348  if (thisisclient) {
349  /* Check if it matches the referenced one.*/
350  if (vallen != strp->length) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
351  urosTcpRosExpect(tcpstp, strp->datap, strp->length);
352  if (tcpstp->err == UROS_OK) { continue; }
353  else { goto _error; }
354  }
355  } else if (_GOT("tcp_nodelay=", 12)) {
356  if (thisisclient) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
357  if (tcpstp->flags.service) { tcpstp->err = UROS_ERR_PARSE; goto _error; }
358  /* tcp_nodelay=(0|1) */
359  if (vallen == 13) {
360  urosTcpRosRecv(tcpstp, buf, 1);
361  if (buf[0] == '1') { tcpstp->flags.noDelay = UROS_TRUE; continue; }
362  else if (buf[0] == '0') { tcpstp->flags.noDelay = UROS_FALSE; continue; }
363  else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
364  } else { tcpstp->err = UROS_ERR_PARSE; goto _error; }
365  }
366  _CHKOK
367 
368  /* Check if the field is already populated.*/
369  urosAssert(strp != NULL);
370  urosError(strp->length > 0 || strp->datap != NULL,
371  { tcpstp->err = UROS_ERR_BADPARAM; goto _error; },
372  ("Field already populated as [%.*s]\n", UROS_STRARG(strp)));
373 
374  /* Allocate and read the value.*/
375  if (vallen > 0) {
376  valp = (char*)urosAlloc(vallen);
377  if (valp == NULL) { tcpstp->err = UROS_ERR_NOMEM; goto _error; }
378  strp->length = vallen;
379  strp->datap = valp;
380  urosTcpRosRecv(tcpstp, valp, vallen); _CHKOK
381  } else {
382  strp->length = 0;
383  strp->datap = NULL;
384  }
385  }
386 
387  if (!thisisclient) {
388  /* Temporarily save the topic information inside the TCPROS status.*/
389  tcpstp->topicp = topicp;
390  }
391 
392  return tcpstp->err = UROS_OK;
393 
394 _error:
395  if (!thisisclient) {
396  urosMsgTypeDelete(typep);
397  urosTopicDelete(topicp);
398  }
399  return tcpstp->err;
400 #undef _CHKOK
401 #undef _GOTPREFIX
402 }

404 uros_err_t uros_tcpserver_processtopicheader(UrosTcpRosStatus *tcpstp) {
405 
406  const UrosListNode *topicnodep;
407  UrosListNode *tcpnodep;
408  UrosTopic *topicp;
409  const UrosMsgType *reftypep;
410 
411  urosAssert(tcpstp != NULL);
412  urosAssert(tcpstp->topicp != NULL);
413 
414  /* The reference topic was temporarily saved inside the TCPROS status.*/
415  topicp = (UrosTopic*)tcpstp->topicp;
416  tcpstp->topicp = NULL;
417 
418  /* Check if the topic is actually published.*/
419  topicnodep = urosTopicListFindByName(&urosNode.status.pubTopicList, &topicp->name);
420  urosError(topicnodep == NULL,
421  { tcpstp->err = UROS_ERR_BADPARAM; goto _finally; },
422  ("Topic [%.*s] not found\n", UROS_STRARG(&topicp->name)));
423  reftypep = ((const UrosTopic *)topicnodep->datap)->typep;
424  urosError(0 != urosStringCmp(&topicp->typep->name, &reftypep->name),
425  { tcpstp->err = UROS_ERR_BADPARAM; goto _finally; },
426  ("Found type [%.*s], expected [%.*s]\n",
427  UROS_STRARG(&topicp->typep->name), UROS_STRARG(&reftypep->name)));
428  urosError(0 != urosStringCmp(&topicp->typep->md5str, &reftypep->md5str),
429  { tcpstp->err = UROS_ERR_BADPARAM; goto _finally; },
430  ("Found MD5 [%.*s], expected [%.*s]\n",
431  UROS_STRARG(&topicp->typep->md5str), UROS_STRARG(&reftypep->md5str)));
432  tcpstp->topicp = (UrosTopic*)topicnodep->datap;
433 
434  /* Add this connection to the active publisher connections list.*/
435  tcpnodep = urosNew(UrosListNode);
436  tcpnodep->datap = tcpstp;
437  tcpnodep->nextp = NULL;
438  urosMutexLock(&urosNode.status.pubTcpListLock);
439  urosListAdd(&urosNode.status.pubTcpList, tcpnodep);
440  urosMutexUnlock(&urosNode.status.pubTcpListLock);
441 
442  tcpstp->err = UROS_OK;
443 _finally:
444  urosMsgTypeDelete((UrosMsgType*)topicp->typep);
445  urosTopicDelete(topicp);
446  return tcpstp->err;
447 }

492 uros_err_t uros_tcpcli_topicsubscription(const UrosString *topicnamep,
493  const UrosAddr *pubaddrp) {
494 
495  uros_err_t err;
496  UrosConn conn;
497  UrosTcpRosStatus tcpst;
498  UrosListNode *topicnodep;
499  UrosTopic *topicp;
500  uros_proc_f handler;
501 
502  urosAssert(urosStringNotEmpty(topicnamep));
503  urosAssert(pubaddrp != NULL);
504 #define _CHKOK { if (err != UROS_OK) { goto _error; } }
505 
506  /* Get topic features.*/
507  urosMutexLock(&urosNode.status.subTopicListLock);
508  topicnodep = urosTopicListFindByName(&urosNode.status.subTopicList,
509  topicnamep);
510  if (topicnodep != NULL) {
511  topicp = (UrosTopic*)topicnodep->datap;
512  urosTopicRefInc(topicp);
513  handler = topicp->procf;
514  }
515  urosMutexUnlock(&urosNode.status.subTopicListLock);
516  urosError(topicnodep == NULL, return UROS_ERR_BADPARAM,
517  ("Topic [%.*s] not found\n", UROS_STRARG(topicnamep)));
518 
519  /* Connect to the publisher.*/
520  urosConnObjectInit(&conn);
521  urosConnCreate(&conn, UROS_PROTO_TCP);
522  err = urosConnConnect(&conn, pubaddrp);
523  urosError(err != UROS_OK, goto _error,
524  ("Error %s while connecting to "UROS_ADDRFMT"\n",
525  urosErrorText(err), UROS_ADDRARG(pubaddrp)));
526  urosTpcRosStatusObjectInit(&tcpst, &conn);
527  tcpst.topicp = topicp;
528  tcpst.flags = topicp->flags;
529 
530  /* Send the TCPROS conenction header.*/
531  err = uros_tcpros_sendheader(&tcpst, UROS_TRUE); _CHKOK
532 
533  /* Receive the TCPROS connection header.*/

1090  *

1092  * handled by a new TCPROS Server thread, allocated from a dedicated
1093  * thread pool. If no threads are free, it waits until one becomes
1094  * available again.
1095  * @see urosThreadPoolStartWorker()
1096  *
1097  * @pre There are no other TCPROS listener threads with the same connection
1098  * port.

1102  * @return
1103  * Error code.
1104  */

1108  UrosAddr locaddr;
1109  UrosConn conn;
1110  (void)data;
1111  (void)err;
1112 
1113  /* Setup the local address.*/
1114  locaddr.port = urosNode.config.tcprosAddr.port;
1115  locaddr.ip.dword = UROS_ANY_IP;
1116

1118  urosConnObjectInit(&conn);
1119  err = urosConnCreate(&conn, UROS_PROTO_TCP);
1120  urosAssert(err == UROS_OK);
1121  err = urosConnBind(&conn, &locaddr);
1122  urosAssert(err == UROS_OK);
1123 
1124  /* Start listening.*/
1125  err = urosConnListen(&conn, UROS_TCPROS_LISTENER_BACKLOG);
1126  urosAssert(err == UROS_OK);
1127  while (UROS_TRUE) {
1128  UrosConn *spawnedp;
1129

1131  spawnedp = urosNew(UrosConn);
1132  urosConnObjectInit(spawnedp);
1133  err = urosConnAccept(&conn, spawnedp);
1134  urosAssert(err == UROS_OK);

1137  urosAssert(err == UROS_OK);
1138 #endif

1140  err = urosThreadPoolStartWorker(&urosNode.status.tcpsvrThdPool,
1141  (void*)spawnedp);
1142 
1143  /* Check if anything went wrong.*/
1144  if (err != UROS_OK && urosConnIsValid(spawnedp)) {
1145  urosConnClose(spawnedp);

1155  *

1194  err = uros_tcpserver_processserviceheader(tcpstp);
1195  if (err == UROS_OK) {
1196  /* FIXME: reuse the topic description for further processing in response,
1197  instead of blocking the registered one.*/
1198  handler = tcpstp->topicp->procf;
1199  }
1200  urosMutexUnlock(&urosNode.status.pubServiceListLock);
1201  if (err != UROS_OK) {

1205  urosTcpRosSend(tcpstp, &length, 1);

1210  }

1219  }

1223  tcpstp->errstr = urosStringCloneZ(urosErrorText(err));
1224  urosTcpRosSendError(tcpstp);
1225  goto _finally;
1226  }
1227  }

1230  err = uros_tcpros_sendheader(tcpstp, UROS_FALSE);

1232

1234  tcpstp->err = UROS_OK;

1247  urosConnClose(csp);
1248  urosTpcRosStatusDelete(tcpstp, UROS_TRUE);
