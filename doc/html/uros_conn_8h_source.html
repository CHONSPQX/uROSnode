57 /**
58  * @brief Makes an IP address packed @em dword.
59  *
60  * @param[in] f1
61  * IP address field 1 (byte 3).
62  * @param[in] f2
63  * IP address field 2 (byte 2).
64  * @param[in] f3
65  * IP address field 3 (byte 1).
66  * @param[in] f4
67  * IP address field 4 (byte 0).
68  * @return
69  * The packed little-endian @em dword IP address.
70  *
71  * @par Example
72  * See @ref conn_ex_ipfmt "UROS_IPFMT example".
73  */
74 #define urosIpDword(f1, f2, f3, f4) \
75  ((((uint32_t)(f1) & 0xFFu) << 24u) | \
76  (((uint32_t)(f2) & 0xFFu) << 16u) | \
77  (((uint32_t)(f3) & 0xFFu) << 8u) | \
78  (((uint32_t)(f4) & 0xFFu) << 0u))
79 
80 /**
81  * @brief Binds to any IP addresses.
82  * @note Packed little-endian @em dword.
83  */
84 #define UROS_ANY_IP urosIpDword(0, 0, 0, 0)
85 
86 /**
87  * @brief Binds to any ports.
88  */
89 #define UROS_ANY_PORT ((uint16_t)0)

96 /**
97  * @brief @p printf() compatible format string for an IP address.
98  *
99  * @par Example
100  * @anchor conn_ex_ipfmt
101  * @code{.c}
102  * UrosIp ip;
103  * ip.dword = urosIpAddr(192, 168, 1, 1);
104  * printf("The IP address is: "UROS_IPFMT, UROS_IPARG(&ip));
105  * @endcode
106  * prints
107  * @verbatim The IP address is: 192.168.1.1@endverbatim
108  */
109 #define UROS_IPFMT "%hhu.%hhu.%hhu.%hhu"
110 
111 /**
112  * @brief @p printf() compatible format string for a connection address.
113  *
114  * @par Example
115  * @anchor conn_ex_addrfmt
116  * @code{.c}
117  * UrosAddr addr;
118  * addr.ip.dword = urosIpAddr(192, 168, 1, 1);
119  * addr.port = 12345;
120  * printf("The TCP/IP address is: "UROS_ADDRFMT, UROS_ADDRARG(&addr));
121  * @endcode
122  * prints
123  * @verbatim The TCP/IP address is: 192.168.1.1:12345@endverbatim
124  */
125 #define UROS_ADDRFMT UROS_IPFMT":%hu"

127 /**
128  * @brief Makes the formatted parameters for an IP address.
129  * @details Generates a list of values to be passed to a variable arguments
130  * function.
131  * @note To be used with the @p UROS_IPFMT format string.
132  *
133  * @param[in] ipp
134  * Pointer to an @p UrosIp descriptor.
135  * @return
136  * List of values for a variable arguments function.
137  *
138  * @par Example
139  * See @ref conn_ex_ipfmt "UROS_IPFMT example".
140  */
141 #define UROS_IPARG(ipp) \
142  ((ipp)->fields.field1), \
143  ((ipp)->fields.field2), \
144  ((ipp)->fields.field3), \
145  ((ipp)->fields.field4)
146 
147 /**
148  * @brief Makes the formatted parameters for a connection address.
149  * @details Generates a list of values to be passed to a variable arguments
150  * function.
151  * @note To be used with the @p UROS_ADDRFMT format string.
152  *
153  * @param[in] addrp

155  * @return
156  * List of values for a variable arguments function.
157  *
158  * @par Example

180 /**
181  * @brief Little-endian IP addres record.
182  */
183 typedef union UrosIp {
184  uint32_t dword; /**< @brief Packed dword.*/
185  struct {
186  uint8_t field4; /**< @brief Address field 4.*/
187  uint8_t field3; /**< @brief Address field 3.*/
188  uint8_t field2; /**< @brief Address field 2.*/
189  uint8_t field1; /**< @brief Address field 1.*/
190  } fields; /**< @brief Individual fields.*/
191  uint8_t bytes[4]; /**< @brief Individual bytes.*/
192 } UrosIp;
