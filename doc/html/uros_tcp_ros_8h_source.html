94  * @param[in,out] tcpstp

117 /**
118  * @brief Generate a variable-length array structure.
119  * @details Generates a structure compatible with @p UrosTcpRosArray, where the
120  * entries pointer is correctly typed.
121  *
122  * @param[in] type
123  * Type of an array entry object. To be valid, a valid pointer
124  * type must be obtained by appending a @p * to @p type.
125  */
126 #define UROS_VARARR(type) \
127  struct { uint32_t length; type *entriesp; }
128 
129 /**

137  * @param[in] index
138  * Index of the involved array entry.
139  * @return
140  * The object value at position @p index.
