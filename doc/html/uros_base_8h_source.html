136 /**
137  * @brief Evaluates an assertion.
138  * @details If the constraint expression does not hold, the system is halted.
139  * @warning An assertion is a @b very strong assumption. If an assertion does
140  * not hold, the @b whole system is halted. This is why it is commonly
141  * used only for very strict constraints, which must always hold.
142  * @note By default it does nothing. The user must provide an @p urosAssert()
143  * macro inside the <tt>urosconf.h</tt> configuration file to achieve
144  * the required behavior with platform-dependent code.
145  * @note Assertions can be disabled inside the <tt>urosconf.h</tt>
146  * configuration file. A global switch is @p UROS_USE_ASSERT.
147  *
148  * @param[in] expr
149  * Assertion constraint expression, which must hold.
150  *
151  * @par Example
152  * @code{.c}
153  * #include <assert.h>
154  *
155  * #define urosAssert(expr) \
156  * assert(expr)
157  *
158  * void load_value(int *resultp) {
159  *
160  * urosAssert(resultp != NULL);
161  *
162  * *resultp = 123;
163  * }
164  * @endcode
165  */
166 # define urosAssert(expr) UROS_NOP
167 #endif /* !UROS_USE_ASSERT || !defined(urosAssert) || defined(__DOXYGEN__) */

173 /**
174  * @brief Checks if an error occurred, and generates a message.
175  * @details If the error condition holds, a formatted error message is
176  * generated, and a statement (even a code block) is executed.
177  * @note By default it simply ignores the formatted message. The user must
178  * provide an @p urosError() macro inside the <tt>urosconf.h</tt>
179  * configuration file to achieve the required behavior with
180  * platform-dependent code.
181  * @note Error messages can be disabled inside the <tt>urosconf.h</tt>
182  * configuration file. A global switch is @p UROS_USE_ERROR_MSG.
183  *
184  * @param[in] when
185  * Error condition, which should not hold for a correct code.
186  * @param[in] action
187  * A statement, or code block, which is executed if the error
188  * condition holds.
189  * @param[in] msgargs
190  * Error message arguments. Must be enclosed between round parentheses
191  * in a @p printf() compatible format.
192  *
193  * @par Example
194  * @code{.c}
195  * #include <urosUser.h>
196  *
197  * #define urosError(when, action, msgargs) \
198  * { if (when) { \
199  * urosUserErrPrintf("Error at %s:%d\n" \
200  * " function: %s\n" \
201  * " reason: %s\n" \
202  * " message: ", \
203  * __FILE__, __LINE__, \
204  * __PRETTY_FUNCTION__, \
205  * #when); \
206  * urosUserErrPrintf msgargs ; \
207  * { action; } } }
208  *
209  * UrosString globalPassword = { 0, NULL };
210  *
211  * uros_bool_t set_password(const UrosString *passwp) {
212  *
213  * urosAssert(urosStringIsValid(passwp));
214  *
215  * urosError(passwp->length < 8, return UROS_FALSE,
216  * ("Password [%.*s] too short: length %zu < 8",
217  * UROS_STRARG(passwp), passwp->length));
218  *
219  * urosStringClean(&globalPassword);
220  * globalPassword = urosStringClone(passwp);
221  * return UROS_TRUE;
222  * }
223  * @endcode
224  * when called with <tt>"F0o_B4r"</tt> as password, prints
225  @verbatim
226  Error at security.c:123
227  function: set_password
228  reason: passwp->length < 8
229  message: Password [F0o_B4r] too short: length 7 < 8
230  @endverbatim
231  */
232 # define urosError(when, action, msgargs) { if (when) { action; } }
233 #endif /* !UROS_USE_ERROR_MSG || !defined(urosError) || defined(__DOXYGEN__) */

235 /**
236  * @brief Allocates a typed object.
237  * @details Allocates a memory chunk on the heap with the size of the provided
238  * type.
239  * @see urosAlloc()
240  *
241  * @param[in] type
242  * Type of the object to be allocated. To be valid, a valid pointer
243  * type must be obtained by appending a @p * to @p type.
244  * @return
245  * The address of the allocated memory chunk, casted to a pointer to
246  * the provided type.
247  * @retval NULL
248  * There is not enough contiguous free memory to allocate a memory
249  * block of the requested size.
250  *
251  * @par Example
252  * @code{.c}
253  * int *valuep;
254  * valuep = urosNew(int);
255  * if (valuep != NULL) {
256  * *valuep = 123;
257  * printf("%d", *valuep);
258  * urosFree(valuep);
259  * }
260  * @endcode
261  */
262 #define urosNew(type) \
263  ((type *)urosAlloc(sizeof(type)))
264 
265 /**
266  * @brief Allocates an array.
267  * @details Allocates a memory chunk which can hold an array of contiguous
268  * objects of the provided type.
269  *
270  * @param[in] n
271  * Number of objects to be allocated.
272  * @param[in] type
273  * Type of the object to be allocated. To be valid, a valid pointer
274  * type must be obtained by appending a @p * to @p type.
275  * @return
276  * The address of the allocated memory chunk, casted to a pointer to
277  * the provided type.
278  * @retval NULL
279  * There is not enough contiguous free memory to allocate a memory
280  * block of the requested size.
281  */
282 #define urosArrayAlloc(n, type) \
283  ((type *)urosAlloc((size_t)(n) * sizeof(type)))
